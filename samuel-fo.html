function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('Transactions') || ss.insertSheet('Transactions'); if (sh.getLastRow() === 0) { sh.appendRow(['ID','Date','Head','Type', 'Payment','Amount','Narration']); sh.getRange(1,1,1,7).setFontWeight('bold') .setBackground('#1a1a1a').setFontColor('#ffffff'); } var d = JSON.parse(e.postData.contents); sh.appendRow([d.id,d.date,d.head,d.type, d.payment,d.amount,d.narration]); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('Transactions'); if (!sh || sh.getLastRow() <= 1) return ContentService.createTextOutput('[]'); var rows = sh.getDataRange().getValues(); var headers = rows[0]; var data = rows.slice(1).map(function(r) { var obj = {}; headers.forEach(function(h,i){ obj[h]=r[i]; }); return obj; }); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
